1//! Reading MARC Authority records from binary streams.
2//!
3//! This module provides specialized reading functionality for MARC Authority records (Type Z).
4//! Authority records use the same ISO 2709 binary format as bibliographic records but are
5//! parsed into [`AuthorityRecord`] structures instead of [`crate::record::Record`] structures.
6//!
7//! # Phase H Integration (Feature Compatibility - Phase I)
8//!
9//! As of Phase I (Feature Compatibility Updates), `AuthorityMarcReader` supports the
10//! unified `ReaderBackend` enum interface enabling:
11//! - **`RustFile`**: Direct file I/O via `std::fs::File` (enables Rayon parallelism)
12//! - **`CursorBackend`**: In-memory reads from bytes via `std::io::Cursor`
13//! - **`PythonFile`**: Python file-like objects (requires GIL, used sequentially)
14//!
15//! Example with `RustFile` for parallel processing:
16//! ```no_run
17//! use mrrc::authority_reader::AuthorityMarcReader;
18//! use std::fs::File;
19//!
20//! let file = File::open("authority_records.mrc")?;
21//! let mut reader = AuthorityMarcReader::new(file);
22//!
23//! while let Some(record) = reader.read_record()? {
24//!     println!("Authority: {}", record.heading_text());
25//! }
26//! # Ok::<(), Box<dyn std::error::Error>>(())
27//! ```
28
29use crate::authority_record::AuthorityRecord;
30use crate::error::{MarcError, Result};
31use crate::leader::Leader;
32use crate::record::Field;
33use crate::recovery::RecoveryMode;
34use std::io::Read;
35
36const FIELD_TERMINATOR: u8 = 0x1E;
37const SUBFIELD_DELIMITER: u8 = 0x1F;
38
39/// Reader for ISO 2709 binary MARC Authority records.
40///
41/// `AuthorityMarcReader` reads Authority record data and converts it to
42/// [`AuthorityRecord`] instances. It reuses the same binary format as bibliographic
43/// records but organizes fields by their functional role (heading, tracings, notes, etc.).
44///
45/// # Backends (Phase H Integration)
46///
47/// The reader automatically detects the input source:
48/// - File paths → `RustFile` backend (parallel-safe)
49/// - Bytes/BytesIO → `CursorBackend` (parallel-safe)
50/// - Python file objects → `PythonFile` (sequential, requires GIL)
51#[derive(Debug)]
52pub struct AuthorityMarcReader<R: Read> {
53    reader: R,
54    recovery_mode: RecoveryMode,
55}
56
57impl<R: Read> AuthorityMarcReader<R> {
58    /// Create a new Authority MARC reader.
59    ///
60    /// # Arguments
61    ///
62    /// * `reader` - Any source implementing [`std::io::Read`]
63    #[must_use]
64    pub fn new(reader: R) -> Self {
65        AuthorityMarcReader {
66            reader,
67            recovery_mode: RecoveryMode::Strict,
68        }
69    }
70
71    /// Set the recovery mode for handling malformed records.
72    ///
73    /// The recovery mode determines how the reader handles truncated or
74    /// malformed Authority records:
75    /// - `Strict`: Return errors immediately (default)
76    /// - `Lenient`: Attempt to recover and salvage valid data
77    /// - `Permissive`: Be very lenient, accepting partial data
78    #[must_use]
79    pub fn with_recovery_mode(mut self, mode: RecoveryMode) -> Self {
80        self.recovery_mode = mode;
81        self
82    }
83
84    /// Read the next Authority record from the stream.
85    ///
86    /// Returns `Ok(None)` when the end of file is reached.
87    ///
88    /// # Errors
89    ///
90    /// Returns an error if the binary data is malformed or an I/O error occurs.
91    #[allow(clippy::too_many_lines)]
92    pub fn read_record(&mut self) -> Result<Option<AuthorityRecord>> {
93        // Read the leader (24 bytes)
94        let mut leader_bytes = vec![0u8; 24];
95        match self.reader.read_exact(&mut leader_bytes) {
96            Ok(()) => {},
97            Err(e) if e.kind() == std::io::ErrorKind::UnexpectedEof => {
98                return Ok(None);
99            },
100            Err(e) => return Err(MarcError::IoError(e)),
101        }
102
103        let leader = Leader::from_bytes(&leader_bytes)?;
104
105        // Verify this is an authority record (Type Z)
106        if leader.record_type != 'z' {
107            return Err(MarcError::InvalidRecord(format!(
108                "Expected authority record type 'z', got '{}'",
109                leader.record_type
110            )));
111        }
112
113        // Calculate directory and data sizes
114        let record_length = leader.record_length as usize;
115        let base_address = leader.data_base_address as usize;
116        let directory_size = base_address - 24;
117
118        // Read record data
119        let mut record_data = vec![0u8; record_length - 24];
120        match self.reader.read_exact(&mut record_data) {
121            Ok(()) => {},
122            Err(e) if e.kind() == std::io::ErrorKind::UnexpectedEof => {
123                if self.recovery_mode == RecoveryMode::Strict {
124                    return Err(MarcError::TruncatedRecord(
125                        "Unexpected end of file while reading record data".to_string(),
126                    ));
127                }
128            },
129            Err(e) => return Err(MarcError::IoError(e)),
130        }
131
132        // Handle truncation in recovery mode
133        if record_data.len() < (record_length - 24) && self.recovery_mode != RecoveryMode::Strict {
134            // For now, treat truncated authority records as a strict error
135            // In the future, we could implement recovery logic
136            return Err(MarcError::TruncatedRecord(
137                "Authority record is truncated".to_string(),
138            ));
139        }
140
141        // Extract directory and data sections
142        let directory_end = std::cmp::min(directory_size, record_data.len());
143        let directory = if directory_end > 0 {
144            &record_data[..directory_end]
145        } else {
146            &[]
147        };
148        let data_start = std::cmp::min(base_address - 24, record_data.len());
149        let data = if data_start < record_data.len() {
150            &record_data[data_start..]
151        } else {
152            &[]
153        };
154        let _ = data; // Will be used when parsing fields
155
156        let mut record = AuthorityRecord::new(leader);
157
158        // Parse directory and extract fields
159        let mut pos = 0;
160        while pos < directory.len() {
161            if directory[pos] == FIELD_TERMINATOR {
162                break;
163            }
164
165            if pos + 12 > directory.len() {
166                if self.recovery_mode == RecoveryMode::Strict {
167                    return Err(MarcError::InvalidRecord(
168                        "Incomplete directory entry".to_string(),
169                    ));
170                }
171                break;
172            }
173
174            // Parse directory entry (12 bytes: tag + length + start position)
175            let tag = std::str::from_utf8(&directory[pos..pos + 3])
176                .map_err(|_| MarcError::InvalidRecord("Invalid field tag".to_string()))?
177                .to_string();
178
179            let length = std::str::from_utf8(&directory[pos + 3..pos + 7])
180                .ok()
181                .and_then(|s| s.parse::<usize>().ok())
182                .ok_or_else(|| MarcError::InvalidRecord("Invalid field length".to_string()))?;
183
184            let start = std::str::from_utf8(&directory[pos + 7..pos + 12])
185                .ok()
186                .and_then(|s| s.parse::<usize>().ok())
187                .ok_or_else(|| {
188                    MarcError::InvalidRecord("Invalid field start position".to_string())
189                })?;
190
191            pos += 12;
192
193            // Extract field data
194            let field_data_start = data_start + start;
195            let field_data_end = std::cmp::min(field_data_start + length, record_data.len());
196
197            if field_data_start >= record_data.len() {
198                continue;
199            }
200
201            let field_bytes = &record_data[field_data_start..field_data_end];
202
203            // Check if this is a control field (00X-009)
204            if tag.len() == 3 && tag.as_str() < "010" {
205                // Control field - no indicators or subfields
206                let value = String::from_utf8_lossy(field_bytes)
207                    .trim_end_matches(['\x1E', '\x1F'])
208                    .to_string();
209                record.add_control_field(tag, value);
210            } else {
211                // Data field - has indicators and subfields
212                if field_bytes.len() < 2 {
213                    continue;
214                }
215
216                let indicator1 = field_bytes[0] as char;
217                let indicator2 = field_bytes[1] as char;
218
219                let mut field = Field {
220                    tag: tag.clone(),
221                    indicator1,
222                    indicator2,
223                    subfields: Vec::new(),
224                };
225
226                // Parse subfields
227                let mut subfield_pos = 2;
228                while subfield_pos < field_bytes.len() {
229                    if field_bytes[subfield_pos] == SUBFIELD_DELIMITER {
230                        if subfield_pos + 1 < field_bytes.len() {
231                            let code = field_bytes[subfield_pos + 1] as char;
232                            subfield_pos += 2;
233
234                            // Find the end of this subfield
235                            let mut subfield_end = subfield_pos;
236                            while subfield_end < field_bytes.len()
237                                && field_bytes[subfield_end] != SUBFIELD_DELIMITER
238                                && field_bytes[subfield_end] != FIELD_TERMINATOR
239                            {
240                                subfield_end += 1;
241                            }
242
243                            let value =
244                                String::from_utf8_lossy(&field_bytes[subfield_pos..subfield_end])
245                                    .to_string();
246                            field
247                                .subfields
248                                .push(crate::record::Subfield { code, value });
249
250                            subfield_pos = subfield_end;
251                        } else {
252                            break;
253                        }
254                    } else if field_bytes[subfield_pos] == FIELD_TERMINATOR {
255                        break;
256                    } else {
257                        subfield_pos += 1;
258                    }
259                }
260
261                // Organize field by its function based on tag
262                match tag.as_str() {
263                    "100" | "110" | "111" | "130" | "148" | "150" | "151" | "155" => {
264                        // 1XX fields - the main heading
265                        record.set_heading(field);
266                    },
267                    "400" | "410" | "411" | "430" | "448" | "450" | "451" | "455" => {
268                        // 4XX fields - see from tracings
269                        record.add_see_from_tracing(field);
270                    },
271                    "500" | "510" | "511" | "530" | "548" | "550" | "551" | "555" => {
272                        // 5XX fields - see also from tracings
273                        record.add_see_also_tracing(field);
274                    },
275                    "660" | "661" | "662" | "663" | "664" | "665" | "666" | "667" | "668"
276                    | "669" | "670" | "671" | "672" | "673" | "674" | "675" | "676" | "677"
277                    | "678" | "679" | "680" | "681" | "682" | "683" | "684" | "685" | "686"
278                    | "687" | "688" | "689" => {
279                        // 66X-68X fields - notes
280                        record.add_note(field);
281                    },
282                    "700" | "710" | "711" | "730" | "748" | "750" | "751" | "755" => {
283                        // 7XX fields - heading linking entries
284                        record.add_linking_entry(field);
285                    },
286                    _ => {
287                        // Other fields
288                        record.add_field(field);
289                    },
290                }
291            }
292        }
293
294        Ok(Some(record))
295    }
296}

342        data.push(b'2');
343        // Base address of data
344        data.extend_from_slice(b"00025");
345        // Encoding level
346        data.push(b' ');
347        // Cataloging form
348        data.push(b'a');
349        // Multipart level

351        // Reserved
352        data.extend_from_slice(b"4500");

354        data.push(FIELD_TERMINATOR);
