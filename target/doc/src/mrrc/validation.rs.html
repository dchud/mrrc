1//! Indicator and field validation for MARC records.
2//!
3//! This module provides validation of MARC field indicators according to MARC21 standards.
4//! Indicators are the two characters following a field tag that specify how the field should
5//! be interpreted.
6
7use crate::error::{MarcError, Result};
8use crate::record::Field;
9use std::collections::HashMap;
10
11/// Validation rules for a field's indicators
12#[derive(Debug, Clone)]
13pub struct IndicatorRules {
14    /// Tag this rule applies to
15    pub tag: String,
16    /// Validation for first indicator
17    pub indicator1: IndicatorValidation,
18    /// Validation for second indicator
19    pub indicator2: IndicatorValidation,
20}
21
22/// Validation rule for a single indicator
23#[derive(Debug, Clone)]
24pub enum IndicatorValidation {
25    /// Indicator is undefined; blank (#) is required
26    Undefined,
27    /// Indicator can be any single character
28    Any,
29    /// Indicator must be one of the specified values
30    Values(Vec<char>),
31    /// Indicator must be a digit (0-9) within the specified range
32    DigitRange {
33        /// Minimum digit value (0-9)
34        min: u8,
35        /// Maximum digit value (0-9)
36        max: u8,
37    },
38    /// Indicator is not defined in current standard (deprecated)
39    Obsolete,
40}
41
42/// Semantic meaning of an indicator value for a specific field
43#[derive(Debug, Clone)]
44pub struct IndicatorMeaning {
45    /// The character value
46    pub value: char,
47    /// Human-readable meaning
48    pub meaning: String,
49}
50
51impl IndicatorValidation {
52    /// Check if the given character is valid for this indicator
53    #[must_use]
54    pub fn is_valid(&self, c: char) -> bool {
55        match self {
56            IndicatorValidation::Undefined => c == '#' || c == ' ',
57            IndicatorValidation::Any | IndicatorValidation::Obsolete => true,
58            IndicatorValidation::Values(values) => values.contains(&c),
59            IndicatorValidation::DigitRange { min, max } => {
60                if let Some(digit) = c.to_digit(10) {
61                    #[allow(clippy::cast_possible_truncation)]
62                    let d = digit as u8;
63                    d >= *min && d <= *max
64                } else {
65                    false
66                }
67            },
68        }
69    }
70}

324    /// Validate a field's indicators
325    ///
326    /// # Errors
327    ///
328    /// Returns `Err` if indicators don't meet validation rules for this field tag.
329    pub fn validate_field(&self, field: &Field) -> Result<()> {
330        self.validate_indicators(&field.tag, field.indicator1, field.indicator2)
331    }
332
333    /// Validate indicators for a specific tag
334    ///
335    /// # Errors
336    ///
337    /// Returns `Err` if indicators don't meet validation rules.
338    pub fn validate_indicators(&self, tag: &str, indicator1: char, indicator2: char) -> Result<()> {
339        if let Some(rules) = self.rules.get(tag) {
340            if !rules.indicator1.is_valid(indicator1) {
341                return Err(MarcError::InvalidField(format!(
342                    "Invalid indicator1 '{}' for field {}: expected {:?}",
343                    indicator1, tag, rules.indicator1
344                )));
345            }
346            if !rules.indicator2.is_valid(indicator2) {
347                return Err(MarcError::InvalidField(format!(
348                    "Invalid indicator2 '{}' for field {}: expected {:?}",
349                    indicator2, tag, rules.indicator2
350                )));
351            }
352        }
353        // Fields without rules are assumed to accept any indicators
354
355        Ok(())
356    }

528    /// Get the semantic meaning of an indicator value for a field
529    ///
530    /// Returns the human-readable meaning if available, or `None` if the indicator
531    /// is semantic or not defined for this field.
532    #[must_use]
533    pub fn get_indicator_meaning(
534        &self,
535        tag: &str,
536        indicator_num: u8,
537        value: char,
538    ) -> Option<String> {
539        if indicator_num != 1 && indicator_num != 2 {
540            return None;
541        }
542
543        self.semantics.get(tag).and_then(|(ind1, ind2)| {
544            if indicator_num == 1 {
545                ind1.get(&value).cloned()
546            } else {
547                ind2.get(&value).cloned()
548            }
549        })
550    }
551
552    /// Get all semantic meanings for a field's indicator
553    ///
554    /// Returns a vector of (value, meaning) pairs for a given field indicator.
555    #[must_use]
556    pub fn get_indicator_meanings(&self, tag: &str, indicator_num: u8) -> Vec<IndicatorMeaning> {
557        if indicator_num != 1 && indicator_num != 2 {
558            return Vec::new();
559        }
560
561        self.semantics
562            .get(tag)
563            .and_then(|(ind1, ind2)| {
564                let map = if indicator_num == 1 { ind1 } else { ind2 };
565                if map.is_empty() {
566                    None
567                } else {
568                    Some(
569                        map.iter()
570                            .map(|(&value, meaning)| IndicatorMeaning {
571                                value,
572                                meaning: meaning.clone(),
573                            })
574                            .collect(),
575                    )
576                }
577            })
578            .unwrap_or_default()
579    }
