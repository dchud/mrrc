1//! Field collection traits for managing MARC record field collections.

3//! This module provides traits that standardize field collection management
4//! across different record types, reducing code duplication.

6use crate::record::Field;
7
8/// A trait for managing a collection of MARC fields.
9///
10/// This trait defines the operations needed to manage field collections
11/// in a consistent way across different record types. Each implementation
12/// manages a specific `Vec<Field>` collection.
13///
14/// # Examples
15///
16/// ```ignore
17/// impl FieldCollection for AuthorityRecord {
18///     fn add_field_to_collection(&mut self, field: Field, name: &str) {
19///         match name {
20///             "see_from" => self.tracings_see_from.push(field),
21///             "see_also" => self.tracings_see_also.push(field),
22///             _ => {}
23///         }
24///     }
25///
26///     fn get_collection(&self, name: &str) -> Option<&[Field]> {
27///         match name {
28///             "see_from" => Some(&self.tracings_see_from),
29///             "see_also" => Some(&self.tracings_see_also),
30///             _ => None
31///         }
32///     }
33/// }
34/// ```
35pub trait FieldCollection {
36    /// Add a field to a named collection.
37    fn add_field_to_collection(&mut self, field: Field, name: &str);
38
39    /// Get a reference to a named collection.
40    fn get_collection(&self, name: &str) -> Option<&[Field]>;
41}
42
43/// Helper to reduce boilerplate for simple field accessor pairs.
44///
45/// This provides a standardized way to implement simple add/get pairs
46/// for field collections.
47#[derive(Debug)]
48pub struct SimpleFieldCollectionHelper;
49
50impl SimpleFieldCollectionHelper {
51    /// Add a field to a mutable vector reference.
52    #[inline]
53    pub fn add_field(collection: &mut Vec<Field>, field: Field) {
54        collection.push(field);
55    }
56
57    /// Get a slice reference from a vector.
58    #[inline]
59    #[must_use]
60    pub fn get_collection(collection: &[Field]) -> &[Field] {
61        collection
62    }
63}
