4//! # MRRC: MARC Rust Crate
5//!
6//! A high-performance Rust library for reading, writing, and manipulating MARC bibliographic
7//! records in the ISO 2709 binary format.
8//!
9//! ## Quick Start
10//!
11//! ### Reading MARC Records
12//!
13//! ```ignore
14//! use mrrc::MarcReader;
15//! use std::fs::File;
16//!
17//! # fn main() -> Result<(), Box<dyn std::error::Error>> {
18//! let file = File::open("records.mrc")?;
19//! let mut reader = MarcReader::new(file);
20//!
21//! while let Some(record) = reader.read_record()? {
22//!     if let Some(title) = record.title() {
23//!         println!("Title: {}", title);
24//!     }
25//! }
26//! # Ok(())
27//! # }
28//! ```
29//!
30//! ### Creating and Writing MARC Records
31//!
32//! ```ignore
33//! use mrrc::{MarcWriter, Record, Field, Leader};
34//!
35//! # fn main() -> Result<(), Box<dyn std::error::Error>> {
36//! let mut record = Record::new(Leader::default());
37//! record.add_control_field("001".to_string(), "12345".to_string());
38//!
39//! let mut field = Field::new("245".to_string(), '1', '0');
40//! field.add_subfield('a', "Test Title".to_string());
41//! record.add_field(field);
42//!
43//! let mut buffer = Vec::new();
44//! {
45//!     let mut writer = MarcWriter::new(&mut buffer);
46//!     writer.write_record(&record)?;
47//! }
48//! # Ok(())
49//! # }
50//! ```
51//!
52//! ### Using Helper Methods
53//!
54//! ```ignore
55//! use mrrc::{Record, Leader, Field};
56//!
57//! let mut record = Record::new(Leader::default());
58//!
59//! let mut field_245 = Field::new("245".to_string(), '1', '0');
60//! field_245.add_subfield('a', "The Great Gatsby".to_string());
61//! field_245.add_subfield('c', "F. Scott Fitzgerald".to_string());
62//! record.add_field(field_245);
63//!
64//! // Use convenience methods
65//! assert_eq!(record.title(), Some("The Great Gatsby"));
66//! assert_eq!(record.is_book(), true);
67//! ```
68//!
69//! ## Modules
70//!
71//! - [`record`] — Core MARC record structures (`Record`, `Field`, `Subfield`)
72//! - [`reader`] — Reading MARC records from binary data streams
73//! - [`writer`] — Writing MARC records to binary format
74//! - [`boundary_scanner`] — Record boundary detection for parallel processing
75//! - [`leader`] — MARC record leader (24-byte header)
76//! - [`json`] — JSON serialization/deserialization
77//! - [`marcjson`] — MARCJSON format (standard JSON-LD format for MARC)
78//! - [`xml`] — XML serialization/deserialization
79//! - [`csv`] — CSV (Comma-Separated Values) export format
80//! - [`dublin_core`] — Dublin Core metadata serialization
81//! - [`mods`] — MODS (Metadata Object Description Schema) serialization
82//! - [`encoding`] — Character encoding support (MARC-8 and UTF-8)
83//! - [`error`] — Error types and result type
84//!
85//! ## Format Support
86//!
87//! The library supports:
88//! - **ISO 2709 Binary Format** — The standard MARC interchange format
89//! - **JSON** — Generic JSON representation of records
90//! - **MARCJSON** — Standard JSON-LD format for MARC records
91//! - **XML** — XML representation with proper field/subfield structure
92//! - **CSV** — Tabular export format for spreadsheet applications
93//! - **Dublin Core** — Simplified metadata schema for discovery
94//! - **MODS** — Detailed metadata description schema for libraries
95//! - **Character Encodings** — MARC-8 and UTF-8 with automatic detection
