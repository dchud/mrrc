1//! Reading MARC records from binary streams.
2//!
3//! This module provides [`MarcReader`] for reading ISO 2709 formatted MARC records
4//! from any source that implements [`std::io::Read`].
5//!
6//! # Examples
7//!
8//! Reading records from a file:
9//!
10//! ```no_run
11//! use mrrc::MarcReader;
12//! use std::fs::File;
13//!
14//! let file = File::open("records.mrc")?;
15//! let mut reader = MarcReader::new(file);
16//!
17//! while let Some(record) = reader.read_record()? {
18//!     println!("Record type: {}", record.leader.record_type);
19//! }
20//! # Ok::<(), Box<dyn std::error::Error>>(())
21//! ```
22//!
23//! Reading from a buffer:
24//!
25//! ```
26//! use mrrc::MarcReader;
27//! use std::io::Cursor;
28//!
29//! let data = b"...binary MARC data...";
30//! let cursor = Cursor::new(data.to_vec());
31//! let mut reader = MarcReader::new(cursor);
32//!
33//! # Ok::<(), Box<dyn std::error::Error>>(())
34//! ```
35

42const FIELD_TERMINATOR: u8 = 0x1E;
43const SUBFIELD_DELIMITER: u8 = 0x1F;

45/// Reader for ISO 2709 binary MARC format.
46///
47/// `MarcReader` reads one MARC record at a time from any source implementing [`std::io::Read`].
48/// Records are fully parsed and returned as [`Record`] instances.
49///
50/// # Examples
51///
52/// ```
53/// use mrrc::MarcReader;
54/// use std::io::Cursor;
55///
56/// let binary_data = vec![]; // MARC binary data
57/// let cursor = Cursor::new(binary_data);
58/// let mut reader = MarcReader::new(cursor);
59///
60/// match reader.read_record() {
61///     Ok(Some(record)) => println!("Record type: {}", record.leader.record_type),
62///     Ok(None) => println!("End of file"),
63///     Err(e) => eprintln!("Error: {}", e),
64/// }
65/// ```
66#[derive(Debug)]
67pub struct MarcReader<R: Read> {
68    reader: R,
69    recovery_mode: RecoveryMode,
70}
71
72impl<R: Read> MarcReader<R> {
73    /// Create a new MARC reader.
74    ///
75    /// # Arguments
76    ///
77    /// * `reader` - Any source implementing [`std::io::Read`]
78    ///
79    /// # Examples
80    ///
81    /// ```
82    /// use mrrc::MarcReader;
83    /// use std::io::Cursor;
84    ///
85    /// let data = vec![];
86    /// let cursor = Cursor::new(data);
87    /// let reader = MarcReader::new(cursor);
88    /// ```
89    pub fn new(reader: R) -> Self {
90        MarcReader {
91            reader,
92            recovery_mode: RecoveryMode::Strict,
93        }
94    }

96    /// Set the recovery mode for handling malformed records.
97    ///
98    /// The recovery mode determines how the reader handles truncated or
99    /// malformed MARC records:
100    /// - `Strict`: Return errors immediately (default)
101    /// - `Lenient`: Attempt to recover and salvage valid data
102    /// - `Permissive`: Be very lenient, accepting partial data
103    ///
104    /// # Examples
105    ///
106    /// ```
107    /// use mrrc::{MarcReader, RecoveryMode};
108    /// use std::io::Cursor;
109    ///
110    /// let data = vec![];
111    /// let cursor = Cursor::new(data);
112    /// let mut reader = MarcReader::new(cursor)
113    ///     .with_recovery_mode(RecoveryMode::Lenient);
114    /// ```
115    #[must_use]
116    pub fn with_recovery_mode(mut self, mode: RecoveryMode) -> Self {
117        self.recovery_mode = mode;
118        self
119    }

121    /// Read a single MARC record.
122    ///
123    /// Returns `Ok(Some(record))` if a record was successfully read, `Ok(None)` if EOF
124    /// was reached, or `Err` if a parsing error occurred.
125    ///
126    /// # Examples
127    ///
128    /// ```
129    /// use mrrc::MarcReader;
130    /// use std::io::Cursor;
131    ///
132    /// # let data = vec![];
133    /// # let cursor = Cursor::new(data);
134    /// let mut reader = MarcReader::new(cursor);
135    ///
136    /// match reader.read_record() {
137    ///     Ok(Some(record)) => { /* process record */ },
138    ///     Ok(None) => println!("End of file"),
139    ///     Err(e) => eprintln!("Error: {}", e),
140    /// }
141    /// ```
142    ///
143    /// # Errors
144    ///
145    /// Returns an error if:
146    /// - The binary data is malformed
147    /// - The record structure is invalid
148    /// - An I/O error occurs
149    #[allow(clippy::too_many_lines, clippy::redundant_else)]
150    pub fn read_record(&mut self) -> Result<Option<Record>> {
151        // Read the leader (24 bytes)
152        let mut leader_bytes = vec![0u8; 24];
153        match self.reader.read_exact(&mut leader_bytes) {
154            Ok(()) => {},
155            Err(e) if e.kind() == std::io::ErrorKind::UnexpectedEof => {
156                // End of file
157                return Ok(None);
158            },
159            Err(e) => return Err(MarcError::IoError(e)),
160        }
161
162        let leader = Leader::from_bytes(&leader_bytes)?;
163
164        // Calculate the size of the directory and data areas
165        let record_length = leader.record_length as usize;
166        let base_address = leader.data_base_address as usize;
167
168        // Directory starts after leader, ends at base_address
169        let directory_size = base_address - 24;
170
171        // Try to read the full record data, but handle truncation gracefully
172        let mut record_data = vec![0u8; record_length - 24];
173        match self.reader.read_exact(&mut record_data) {
174            Ok(()) => {},
175            Err(e) if e.kind() == std::io::ErrorKind::UnexpectedEof => {
176                // Record is truncated
177                if self.recovery_mode == RecoveryMode::Strict {
178                    return Err(MarcError::TruncatedRecord(
179                        "Unexpected end of file while reading record data".to_string(),
180                    ));
181                }
182                // For lenient/permissive modes, attempt recovery with partial data
183            },
184            Err(e) => return Err(MarcError::IoError(e)),
185        }
186
187        // If the record is truncated and we're in recovery mode, use recovery logic
188        if record_data.len() < (record_length - 24) && self.recovery_mode != RecoveryMode::Strict {
189            return crate::recovery::try_recover_record(
190                leader,
191                &record_data,
192                base_address,
193                self.recovery_mode,
194            )
195            .map(Some);
196        }
197
198        // Parse directory
199        let directory_end = std::cmp::min(directory_size, record_data.len());
200        let directory = if directory_end > 0 {
201            &record_data[..directory_end]
202        } else {
203            &[]
204        };
205        let data_start = std::cmp::min(base_address - 24, record_data.len());
206        let data = if data_start < record_data.len() {
207            &record_data[data_start..]
208        } else {
209            &[]
210        };
211
212        let mut record = Record::new(leader);
213
214        // Parse directory entries (12 bytes each: tag(3) + length(4) + start position(5))
215        // Directory is terminated with FIELD_TERMINATOR
216        let mut pos = 0;
217        while pos < directory.len() {
218            if directory[pos] == FIELD_TERMINATOR {
219                // End of directory
220                break;
221            }
222
223            if pos + 12 > directory.len() {
224                if self.recovery_mode == RecoveryMode::Strict {
225                    return Err(MarcError::InvalidRecord(
226                        "Incomplete directory entry".to_string(),
227                    ));
228                } else {
229                    break;
230                }
231            }
232
233            let entry_chunk = &directory[pos..pos + 12];
234            let tag = String::from_utf8_lossy(&entry_chunk[0..3]).to_string();
235            let field_length = match parse_4digits(&entry_chunk[3..7]) {
236                Ok(len) => len,
237                Err(e) => {
238                    if self.recovery_mode == RecoveryMode::Strict {
239                        return Err(e);
240                    } else {
241                        pos += 12;
242                        continue;
243                    }
244                },
245            };
246            let start_position = match parse_digits(&entry_chunk[7..12]) {
247                Ok(pos) => pos,
248                Err(e) => {
249                    if self.recovery_mode == RecoveryMode::Strict {
250                        return Err(e);
251                    } else {
252                        pos += 12;
253                        continue;
254                    }
255                },
256            };
257            pos += 12;
258
259            let end_position = start_position + field_length;
260            if end_position > data.len() {
261                if self.recovery_mode == RecoveryMode::Strict {
262                    return Err(MarcError::InvalidRecord(format!(
263                        "Field {tag} exceeds data area"
264                    )));
265                } else {
266                    // Try to read what we have
267                    let available_end = std::cmp::min(end_position, data.len());
268                    if available_end > start_position {
269                        let field_data = &data[start_position..available_end];
270                        if tag != "LDR" {
271                            if tag.starts_with('0')
272                                && tag.chars().all(char::is_numeric)
273                                && tag.as_str() < "010"
274                            {
275                                let value = String::from_utf8_lossy(
276                                    &field_data[..field_data.len().saturating_sub(1)],
277                                )
278                                .to_string();
279                                record.add_control_field(tag, value);
280                            } else if let Ok(field) = parse_data_field(field_data, &tag) {
281                                record.add_field(field);
282                            }
283                        }
284                    }
285                    continue;
286                }
287            }
288
289            let field_data = &data[start_position..end_position];
290
291            // Parse field (skip LDR as it's already parsed)
292            if tag != "LDR" {
293                if tag.starts_with('0') && tag.chars().all(char::is_numeric) && tag.as_str() < "010"
294                {
295                    // Control field (001-009)
296                    let value = String::from_utf8_lossy(
297                        &field_data[..field_data.len().saturating_sub(1)], // Remove field terminator
298                    )
299                    .to_string();
300                    record.add_control_field(tag, value);
301                } else {
302                    // Data field (010+)
303                    match parse_data_field(field_data, &tag) {
304                        Ok(field) => record.add_field(field),
305                        Err(e) => {
306                            if self.recovery_mode == RecoveryMode::Strict {
307                                return Err(MarcError::InvalidField(format!("Tag {tag}: {e}")));
308                            }
309                            // In lenient/permissive mode, skip this field and continue
310                        },
311                    }
312                }
313            }
314        }
315
316        Ok(Some(record))
317    }
318}

320/// Parse a data field from raw bytes
321fn parse_data_field(data: &[u8], tag: &str) -> Result<Field> {
322    if data.len() < 2 {
323        return Err(MarcError::InvalidField(
324            "Data field too short (needs indicators)".to_string(),
325        ));
326    }
327
328    let indicator1 = data[0] as char;
329    let indicator2 = data[1] as char;
330    let mut field = Field::new(tag.to_string(), indicator1, indicator2);
331
332    // Parse subfields
333    let subfield_data = &data[2..];
334    let mut current_position = 0;
335
336    while current_position < subfield_data.len() {
337        if subfield_data[current_position] == FIELD_TERMINATOR {
338            // End of field
339            break;
340        }
341
342        if subfield_data[current_position] == SUBFIELD_DELIMITER {
343            current_position += 1;
344            if current_position >= subfield_data.len() {
345                break;
346            }
347
348            let code = subfield_data[current_position] as char;
349            current_position += 1;
350
351            // Find next subfield or field terminator
352            let mut end = current_position;
353            while end < subfield_data.len()
354                && subfield_data[end] != SUBFIELD_DELIMITER
355                && subfield_data[end] != FIELD_TERMINATOR
356            {
357                end += 1;
358            }
359
360            let value = String::from_utf8_lossy(&subfield_data[current_position..end]).to_string();
361            field.add_subfield(code, value);
362            current_position = end;
363        } else {
364            return Err(MarcError::InvalidField(
365                "Expected subfield delimiter".to_string(),
366            ));
367        }
368    }
369
370    Ok(field)
371}

373/// Parse a 4-digit ASCII number from bytes
374fn parse_4digits(bytes: &[u8]) -> Result<usize> {
375    if bytes.len() != 4 {
376        return Err(MarcError::InvalidRecord(format!(
377            "Expected 4-digit field, got {} bytes",
378            bytes.len()
379        )));
380    }
381
382    let s = String::from_utf8_lossy(bytes);
383    s.parse::<usize>()
384        .map_err(|_| MarcError::InvalidRecord(format!("Invalid numeric field: '{s}'")))
385}
386
387/// Parse a 5-digit ASCII number from bytes
388fn parse_digits(bytes: &[u8]) -> Result<usize> {
389    if bytes.len() != 5 {
390        return Err(MarcError::InvalidRecord(format!(
391            "Expected 5-digit field, got {} bytes",
392            bytes.len()
393        )));
394    }
395
396    let s = String::from_utf8_lossy(bytes);
397    s.parse::<usize>()
398        .map_err(|_| MarcError::InvalidRecord(format!("Invalid numeric field: '{s}'")))
399}

408    #[test]
409    fn test_read_simple_record() {
410        // Manually build a valid MARC record
411        let mut record_bytes = Vec::new();
412
413        // Data area: field 245
414        let mut field_245 = Vec::new();
415        field_245.extend_from_slice(b"10"); // Indicators
416        field_245.push(SUBFIELD_DELIMITER);
417        field_245.push(b'a');
418        field_245.extend_from_slice(b"Test title");
419        field_245.push(FIELD_TERMINATOR);
420
421        // Directory (without terminator yet)
422        let mut directory = Vec::new();
423        directory.extend_from_slice(b"245");
424        directory.extend_from_slice(format!("{:04}", field_245.len()).as_bytes());
425        directory.extend_from_slice(b"00000");
426
427        // Base address is after leader + directory + directory terminator
428        let base_address = 24 + directory.len() + 1; // +1 for directory terminator
429        directory.push(FIELD_TERMINATOR);
430        let record_length = base_address + field_245.len() + 1;
431
432        // Leader (must be exactly 24 bytes)
433        let mut leader = Vec::new();
434        leader.extend_from_slice(format!("{record_length:05}").as_bytes()); // 0-4
435        leader.push(b'n'); // 5: status
436        leader.push(b'a'); // 6: type
437        leader.push(b'm'); // 7: bib level
438        leader.push(b' '); // 8: control type
439        leader.push(b'a'); // 9: character coding
440        leader.push(b'2'); // 10: indicator count
441        leader.push(b'2'); // 11: subfield code count
442        leader.extend_from_slice(format!("{base_address:05}").as_bytes()); // 12-16
443        leader.push(b' '); // 17: encoding level
444        leader.push(b' '); // 18: cataloging form
445        leader.push(b' '); // 19: multipart level
446        leader.extend_from_slice(b"4500"); // 20-23: reserved
447
448        // Assemble
449        record_bytes.extend_from_slice(&leader);
450        record_bytes.extend_from_slice(&directory);
451        record_bytes.extend_from_slice(&field_245);
452        record_bytes.push(RECORD_TERMINATOR);
453
454        let cursor = Cursor::new(record_bytes);
455        let mut reader = MarcReader::new(cursor);
456
457        let record = reader.read_record().unwrap().unwrap();
458
459        assert_eq!(record.leader.record_type, 'a');
460        let fields = record.get_fields("245");
461        assert!(fields.is_some());
462        let field = &fields.unwrap()[0];
463        assert_eq!(field.indicator1, '1');
464        assert_eq!(field.indicator2, '0');
465
466        let title = field.get_subfield('a');
467        assert_eq!(title, Some("Test title"));
468    }
