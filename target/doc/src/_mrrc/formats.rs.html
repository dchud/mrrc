1// Python bindings for MARC format conversion functions
2//
3// This module exposes the format conversion capabilities of mrrc to Python:
4// - JSON serialization/deserialization
5// - XML serialization/deserialization
6// - MARCJSON serialization/deserialization
7// - Dublin Core conversion
8// - MODS XML conversion
9// - CSV export

11use crate::error::marc_error_to_py_err;

13use mrrc::{dublin_core, json, marcjson, mods, xml};
14use pyo3::exceptions::PyValueError;

18/// Convert a MARC record to JSON.
19///
20/// # Arguments
21/// * `record` - A PyRecord instance
22///
23/// # Returns
24/// A JSON string representation of the record
25///
26/// # Example
27/// ```python
28/// import mrrc
29/// record = mrrc.Record(mrrc.Leader())
30/// json_str = mrrc.record_to_json(record)
31/// ```
32#[pyfunction]
33pub fn record_to_json(record: &PyRecord) -> PyResult<String> {
34    json::record_to_json(&record.inner)
35        .map(|v| v.to_string())
36        .map_err(marc_error_to_py_err)
37}
38
39/// Convert JSON back to a MARC record.
40///
41/// # Arguments
42/// * `json_str` - A JSON string representing a MARC record
43///
44/// # Returns
45/// A PyRecord instance
46///
47/// # Example
48/// ```python
49/// import mrrc
50/// json_str = '...'  # JSON representation of a record
51/// record = mrrc.json_to_record(json_str)
52/// ```
53#[pyfunction]
54pub fn json_to_record(json_str: &str) -> PyResult<PyRecord> {
55    let json_value: Value = serde_json::from_str(json_str)
56        .map_err(|e| PyValueError::new_err(format!("Invalid JSON: {}", e)))?;
57    json::json_to_record(&json_value)
58        .map(|inner| PyRecord { inner })
59        .map_err(marc_error_to_py_err)
60}

146/// Convert a MARC record to Dublin Core metadata.
147///
148/// Returns a dictionary with Dublin Core elements extracted from the record.
149///
150/// # Arguments
151/// * `record` - A PyRecord instance
152///
153/// # Returns
154/// A dictionary mapping Dublin Core field names to values
155///
156/// # Example
157/// ```python
158/// import mrrc
159/// record = mrrc.Record(mrrc.Leader())
160/// dc = mrrc.record_to_dublin_core(record)
161/// print(dc['title'])
162/// ```
163#[pyfunction]
164pub fn record_to_dublin_core(
165    record: &PyRecord,
166) -> PyResult<std::collections::HashMap<String, Vec<String>>> {
167    dublin_core::record_to_dublin_core(&record.inner)
168        .map(|dc| {
169            let mut map = std::collections::HashMap::new();
170            map.insert("title".to_string(), dc.title);
171            map.insert("creator".to_string(), dc.creator);
172            map.insert("subject".to_string(), dc.subject);
173            map.insert("description".to_string(), dc.description);
174            map.insert("publisher".to_string(), dc.publisher);
175            map.insert("contributor".to_string(), dc.contributor);
176            map.insert("date".to_string(), dc.date);
177            map.insert("type".to_string(), dc.dc_type);
178            map.insert("format".to_string(), dc.format);
179            map.insert("identifier".to_string(), dc.identifier);
180            map.insert("source".to_string(), dc.source);
181            map.insert("language".to_string(), dc.language);
182            map.insert("relation".to_string(), dc.relation);
183            map.insert("coverage".to_string(), dc.coverage);
184            map.insert("rights".to_string(), dc.rights);
185            map
186        })
187        .map_err(marc_error_to_py_err)
188}

212/// Convert Dublin Core metadata to XML.
213///
214/// # Arguments
215/// * `dublin_core` - A dictionary with Dublin Core elements
216///
217/// # Returns
218/// An XML string in Dublin Core XML format
219///
220/// # Example
221/// ```python
222/// import mrrc
223/// dc = {'title': ['Example'], 'creator': ['Author']}
224/// dc_xml = mrrc.dublin_core_to_xml(dc)
225/// ```
226#[pyfunction]
227pub fn dublin_core_to_xml(
228    dublin_core: std::collections::HashMap<String, Vec<String>>,
229) -> PyResult<String> {
230    // Helper to extract list from map, defaulting to empty vec
231    let get_string_list =
232        |key: &str| -> Vec<String> { dublin_core.get(key).cloned().unwrap_or_default() };
233
234    let dc_record = mrrc::dublin_core::DublinCoreRecord {
235        title: get_string_list("title"),
236        creator: get_string_list("creator"),
237        subject: get_string_list("subject"),
238        description: get_string_list("description"),
239        publisher: get_string_list("publisher"),
240        contributor: get_string_list("contributor"),
241        date: get_string_list("date"),
242        dc_type: get_string_list("type"),
243        format: get_string_list("format"),
244        identifier: get_string_list("identifier"),
245        source: get_string_list("source"),
246        language: get_string_list("language"),
247        relation: get_string_list("relation"),
248        coverage: get_string_list("coverage"),
249        rights: get_string_list("rights"),
250    };
251
252    Ok(dublin_core::dublin_core_to_xml(&dc_record))
253}
