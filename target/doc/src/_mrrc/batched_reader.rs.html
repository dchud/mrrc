1// Batched MARC reader with queue-based state machine for efficient GIL management
2//
3// This module implements Phase C (Batch Reading) of the GIL Release Hybrid Plan.
4// It reads multiple MARC records in a single Python GIL acquisition, then serves
5// them to the caller from an internal queue. This dramatically reduces GIL
6// acquire/release overhead (from N per-record acquisitions to N/100 batch acquisitions).
7//
8// Key design:
9// - VecDeque<SmallVec> for O(1) front-pop during iteration
10// - Hard limits: 200 records or 300KB per batch (prevents unbounded allocation)
11// - EOF state machine ensures idempotent behavior
12// - Single GIL acquire/release cycle per batch (100x reduction vs per-record)
13
14use crate::buffered_reader::BufferedMarcReader;
15use crate::parse_error::ParseError;
16use pyo3::prelude::*;
17use smallvec::SmallVec;
18use std::collections::VecDeque;
19
20/// State machine for batched MARC reading with GIL management
21///
22/// Implements the queue-based state machine defined in GIL_RELEASE_HYBRID_IMPLEMENTATION_PLAN_REVISIONS.md:
23/// INITIAL → CHECK_QUEUE_NON_EMPTY → CHECK_EOF_STATE → READ_BATCH → (repeat)
24///
25/// Design:
26/// - record_queue: VecDeque for O(1) front-pop (serves buffered records)
27/// - buffered_reader: Underlying file reader (for read_batch calls)
28/// - eof_reached: Idempotent EOF flag (prevents redundant I/O after EOF)
29/// - batch_size: Fixed at 100 (subject to validation in C.Gate task)
30#[derive(Debug)]
31pub struct BatchedMarcReader {
32    /// Underlying buffered reader for reading from Python file object
33    buffered_reader: BufferedMarcReader,
34
35    /// Queue of record bytes, ready for parsing
36    /// Using VecDeque for O(1) pop_front() performance
37    record_queue: VecDeque<SmallVec<[u8; 4096]>>,
38
39    /// Cumulative capacity tracking (bytes in queue)
40    queue_capacity_bytes: usize,
41
42    /// EOF flag - once set, reading stops and __next__ returns None idempotently
43    eof_reached: bool,
44
45    /// Target batch size for read_batch() calls
46    /// Fixed at 100 for initial implementation (validated in C.Gate)
47    batch_size: usize,
48}
49
50impl BatchedMarcReader {
51    /// Create a new BatchedMarcReader from a Python file-like object
52    ///
53    /// # Arguments
54    /// * `file_obj` - A Python file-like object supporting .read(n)
55    pub fn new(file_obj: Py<PyAny>) -> Self {
56        BatchedMarcReader {
57            buffered_reader: BufferedMarcReader::new(file_obj),
58            record_queue: VecDeque::new(),
59            queue_capacity_bytes: 0,
60            eof_reached: false,
61            batch_size: 100,
62        }
63    }
64
65    /// Read the next record bytes from the queue or file
66    ///
67    /// Implements state machine:
68    /// 1. If queue not empty: pop and return
69    /// 2. If EOF reached: return None (idempotent)
70    /// 3. Otherwise: read_batch() and process
71    ///
72    /// Returns:
73    /// - Ok(Some(bytes)) - A complete MARC record as owned bytes
74    /// - Ok(None) - End of file reached (subsequent calls also return None)
75    /// - Err(ParseError) - I/O or parsing boundary error
76    pub fn read_next_record_bytes(
77        &mut self,
78        py: Python<'_>,
79    ) -> Result<Option<Vec<u8>>, ParseError> {
80        // === STATE: CHECK_QUEUE_NON_EMPTY ===
81        if !self.record_queue.is_empty() {
82            if let Some(record) = self.record_queue.pop_front() {
83                self.queue_capacity_bytes = self.queue_capacity_bytes.saturating_sub(record.len());
84                return Ok(Some(record.to_vec()));
85            }
86        }
87
88        // === STATE: CHECK_EOF_STATE ===
89        if self.eof_reached {
90            // Idempotent: subsequent calls return None without I/O
91            return Ok(None);
92        }
93
94        // === STATE: READ_BATCH ===
95        let batch = self.read_batch(py)?;
96
97        if batch.is_empty() {
98            // EOF reached during batch read
99            self.eof_reached = true;
100            return Ok(None);
101        }
102
103        // Non-empty batch: extend queue and pop first record
104        for record in batch {
105            self.queue_capacity_bytes = self.queue_capacity_bytes.saturating_add(record.len());
106            self.record_queue.push_back(record);
107        }
108
109        // Re-check queue (should not be empty now)
110        if let Some(record) = self.record_queue.pop_front() {
111            self.queue_capacity_bytes = self.queue_capacity_bytes.saturating_sub(record.len());
112            Ok(Some(record.to_vec()))
113        } else {
114            // Defensive: should never happen if batch was non-empty
115            Ok(None)
116        }
117    }
118
119    /// Read a batch of records with a single GIL acquire/release cycle
120    ///
121    /// # GIL Contract
122    /// - Entry: py parameter guarantees GIL is held
123    /// - Exit: GIL is released when this function returns
124    /// - The closure returns Rust errors only; PyErr conversion deferred to caller
125    ///
126    /// # Batch Size Semantics
127    /// - Target: 100 records per batch
128    /// - Hard limits:
129    ///   - 200 records maximum (prevents unbounded allocation)
130    ///   - 300 KB maximum (prevents memory spikes on large records)
131    /// - If limits are hit, returns partial batch (variable size)
132    /// - Caller continues with next read_batch() call
133    ///
134    /// # Returns
135    /// Vec<SmallVec<[u8; 4096]>> - Records read in this batch (may be < batch_size)
136    /// Empty vec indicates EOF (no more records available)
137    fn read_batch(&mut self, py: Python<'_>) -> Result<Vec<SmallVec<[u8; 4096]>>, ParseError> {
138        let mut batch = Vec::with_capacity(self.batch_size);
139        let mut bytes_read = 0usize;
140
141        // Read up to batch_size records, respecting hard limits
142        while batch.len() < self.batch_size {
143            // Hard limit: 200 records or 300KB
144            if batch.len() >= 200 || bytes_read > 300_000 {
145                break;
146            }
147
148            match self.buffered_reader.read_next_record_bytes(py) {
149                Ok(Some(record_bytes)) => {
150                    bytes_read = bytes_read.saturating_add(record_bytes.len());
151                    batch.push(SmallVec::from_slice(&record_bytes));
152                },
153                Ok(None) => {
154                    // EOF reached during batch read
155                    break;
156                },
157                Err(e) => {
158                    // I/O or boundary error
159                    return Err(e);
160                },
161            }
162        }
163
164        Ok(batch)
165    }
166
167    /// Check if the reader has reached EOF
168    ///
169    /// Returns true after EOF is set (idempotent).
170    #[allow(dead_code)]
171    pub fn is_eof(&self) -> bool {
172        self.eof_reached
173    }
174
175    /// Get current queue size (for diagnostics/testing)
176    #[allow(dead_code)]
177    pub fn queue_len(&self) -> usize {
178        self.record_queue.len()
179    }
180
181    /// Get current queue capacity in bytes (for diagnostics/testing)
182    #[allow(dead_code)]
183    pub fn queue_capacity_bytes(&self) -> usize {
184        self.queue_capacity_bytes
185    }
186}

192    #[test]
193    fn test_batched_reader_queue_operations() {
194        // Create minimal records for testing
195        let r1: SmallVec<[u8; 4096]> = SmallVec::from_slice(b"record1");
196        let r2: SmallVec<[u8; 4096]> = SmallVec::from_slice(b"record2");
197
198        let mut queue = VecDeque::new();
199        queue.push_back(r1);
200        queue.push_back(r2);
201
202        assert_eq!(queue.len(), 2);
203
204        let popped = queue.pop_front();
205        assert!(popped.is_some());
206        assert_eq!(popped.unwrap().as_slice(), b"record1");
207        assert_eq!(queue.len(), 1);
208    }

210    #[test]
211    fn test_queue_capacity_tracking() {
212        let r1: SmallVec<[u8; 4096]> = SmallVec::from_slice(b"record1");
213        let r2: SmallVec<[u8; 4096]> = SmallVec::from_slice(b"record2222");
214
215        let mut capacity = 0usize;
216
217        capacity = capacity.saturating_add(r1.len());
218        assert_eq!(capacity, 7);
219
220        capacity = capacity.saturating_add(r2.len());
221        assert_eq!(capacity, 17);
222
223        capacity = capacity.saturating_sub(r1.len());
224        assert_eq!(capacity, 10);
225    }
