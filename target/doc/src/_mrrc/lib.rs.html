27/// Initialize the Python module
28#[pymodule]
29fn _mrrc(_py: Python<'_>, m: &Bound<'_, PyModule>) -> PyResult<()> {
30    m.add_class::<PyLeader>()?;
31    m.add_class::<PySubfield>()?;
32    m.add_class::<PyField>()?;
33    m.add_class::<PyRecord>()?;
34    m.add_class::<PyMARCReader>()?;
35    m.add_class::<PyMARCWriter>()?;
36    m.add_class::<PyRecordBoundaryScanner>()?;
37    m.add_class::<PyProducerConsumerPipeline>()?;
38
39    // Format conversion functions
40    m.add_function(wrap_pyfunction!(formats::record_to_json, m)?)?;
41    m.add_function(wrap_pyfunction!(formats::json_to_record, m)?)?;
42    m.add_function(wrap_pyfunction!(formats::record_to_xml, m)?)?;
43    m.add_function(wrap_pyfunction!(formats::xml_to_record, m)?)?;
44    m.add_function(wrap_pyfunction!(formats::record_to_marcjson, m)?)?;
45    m.add_function(wrap_pyfunction!(formats::marcjson_to_record, m)?)?;
46    m.add_function(wrap_pyfunction!(formats::record_to_dublin_core, m)?)?;
47    m.add_function(wrap_pyfunction!(formats::record_to_mods, m)?)?;
48    m.add_function(wrap_pyfunction!(formats::dublin_core_to_xml, m)?)?;
49
50    // Rayon parser pool (H.4b) functions
51    m.add_function(wrap_pyfunction!(parse_batch_parallel, m)?)?;
52    m.add_function(wrap_pyfunction!(parse_batch_parallel_limited, m)?)?;
53
54    m.add(
55        "__doc__",
56        "MRRC: A fast MARC library written in Rust with Python bindings",
57    )?;
58    m.add("__version__", "0.1.0")?;
59
60    Ok(())
61}
